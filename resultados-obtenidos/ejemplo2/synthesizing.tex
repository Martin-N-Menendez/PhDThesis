	\begin{lstlisting}[language = {}, caption = Cantidad de elementos a implementar por el ACG, label = {lst:EJ2_8}]
	n_netElements:7
	n_switch:3
	n_doubleSwitch:0
	n_borders:4
	n_buffers:1
	n_levelCrossings:1
	n_platforms:2
	n_scissorCrossings:0
