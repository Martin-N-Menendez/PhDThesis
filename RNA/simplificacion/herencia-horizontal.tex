    \begin{algorithm}[hbt!]
        \caption{Horizontal inheritance algorithm}\label{alg:horizontal}
        \DontPrintSemicolon
        %\SetAlgoLined
        \SetNoFillComment
        \LinesNotNumbered 
        \For{ signal in \{signals\} }
        {
            \If{ each object in netElement }
            {
                \If{ node is between switches with not enough space }
                {
                    Move\_backwards\_in\_main\_line(signal)
                }
                \Else 
                {
                    combine\_signals( )
                }
            }
        }
        \KwResult{[Signals]} 
    \end{algorithm}
